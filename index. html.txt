    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        // Ícones em SVG para evitar dependências externas que causam erro
        const IconShield = ({size=24, className=""}) => (
            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5" strokeLinecap="round" strokeLinejoin="round" className={className}>
                <path d="M12 22s8-4 8-10V5l-8-3-8 3v7c0 6 8 10 8 10z"/>
            </svg>
        );

        const IconFolder = ({size=20}) => (
            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
                <path d="M22 19a2 2 0 0 1-2 2H4a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h5l2 3h9a2 2 0 0 1 2 2z"/>
            </svg>
        );

        const IconTrash = ({size=18}) => (
            <svg width={size} height={size} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
                <polyline points="3 6 5 6 21 6"/><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"/>
            </svg>
        );

        const firebaseConfig = {
            authDomain: "respeito-capital-humano-df.firebaseapp.com",
            projectId: "respeito-capital-humano-df",
            storageBucket: "respeito-capital-humano-df.firebasestorage.app",
            messagingSenderId: "1062425708099",
            appId: "1:1062425708099:web:260cfa86f5112a5b3cf6f6"
        };

        if (!firebase.apps.length) { firebase.initializeApp(firebaseConfig); }
        const db = firebase.firestore();
        const DATA_PATH = 'artifacts/respeito-capital-humano-df/public/data';

        const App = () => {
            const [view, setView] = useState('welcome');
            const [companies, setCompanies] = useState([]);
            const [reports, setReports] = useState([]);
            const [folders, setFolders] = useState([]);
            const [search, setSearch] = useState('');
            const [activeCompany, setActiveCompany] = useState(null);
            const [adminPass, setAdminPass] = useState('');
            const [loading, setLoading] = useState(false);
            const [adminTab, setAdminTab] = useState('reports');
            const [protocol, setProtocol] = useState('');
            const [qrComp, setQrComp] = useState(null);
            const [isIdentified, setIsIdentified] = useState(false);
            const [expandedEmp, setExpandedEmp] = useState(null);

            useEffect(() => {
                firebase.auth().signInAnonymously().then(() => {
                    db.collection(`${DATA_PATH}/companies`).onSnapshot(s => setCompanies(s.docs.map(d => ({id: d.id, ...d.data()}))));
                    db.collection(`${DATA_PATH}/reports`).onSnapshot(s => setReports(s.docs.map(d => ({id: d.id, ...d.data()}))));
                    db.collection(`${DATA_PATH}/folders`).onSnapshot(s => setFolders(s.docs.map(d => ({id: d.id, ...d.data()}))));
                });
            }, []);

            const filteredResults = useMemo(() => {
                const t = search.toLowerCase().trim();
                return t.length >= 2 ? companies.filter(c => c.name.toLowerCase().includes(t)) : [];
            }, [search, companies]);

            const handleReport = async (e) => {
                e.preventDefault();
                setLoading(true);
                const fd = new FormData(e.target);
                const prot = 'RCH-' + Math.random().toString(36).substring(2, 7).toUpperCase();
                try {
                    await db.collection(`${DATA_PATH}/reports`).add({
                        protocolo: prot,
                        empresaName: activeCompany.name,
                        empresaId: activeCompany.id,
                        unidade: fd.get('unidade'),
                        dataOcorrencia: fd.get('dataOcorrencia'),
                        identidade: isIdentified ? 'IDENTIFICADO' : 'ANÓNIMO',
                        nomeRelator: fd.get('nomeRelator') || 'CONFIDENCIAL',
                        relato: fd.get('relato'),
                        contato: fd.get('contato') || 'NENHUM',
                        status: 'ABERTO',
                        folderId: 'default',
                        dataProtocolo: new Date().toISOString()
                    });
                    setProtocol(prot);
                    setView('success');
                } catch (e) { alert("Erro de rede."); }
                setLoading(false);
            };

            const updateRep = async (id, data) => {
                await db.collection(`${DATA_PATH}/reports`).doc(id).update(data);
            };

            const genQR = (c) => {
                setQrComp(c);
                setTimeout(() => {
                    document.getElementById("qr-area").innerHTML = "";
                    new QRCode(document.getElementById("qr-area"), { text: window.location.href, width: 180, height: 180 });
                }, 200);
            };

            // --- TELAS ---

            if (view === 'welcome') return (
                <div className="min-h-screen flex items-center justify-center p-6 animate-fade bg-slate-900">
                    <div className="max-w-xl w-full glass-card text-center">
                        <div className="flex justify-center gap-2 mb-8">
                            <div className="selo-nr1">NR-1 VALIDADA</div>
                            <div className="bg-blue-600 text-white px-3 py-1.5 rounded-lg text-[10px] font-black">CERTIFICADO RCH</div>
                        </div>
                        <IconShield size={54} className="mx-auto text-blue-600 mb-6" />
                        <h1 className="text-2xl font-black uppercase italic tracking-tighter text-slate-900 mb-4">Respeito ao Capital Humano</h1>
                        <div className="bg-slate-50 p-6 rounded-2xl mb-10 border-l-4 border-emerald-600 text-left text-[11px] text-slate-500 italic">
                            "NR-1 Item 1.4.1 g): O empregador deve estabelecer mecanismos para que os trabalhadores comuniquem riscos."
                        </div>
                        <div className="text-left space-y-4">
                            <label className="text-[10px] font-black uppercase text-slate-400 ml-2">Pesquisa de Empresa Cliente:</label>
                            <input type="text" placeholder="Digite o nome..." className="input-rch shadow-sm" value={search} onChange={e => setSearch(e.target.value)} />
                            <div className="space-y-2 mt-2">
                                {filteredResults.map(c => (
                                    <button key={c.id} onClick={() => { setActiveCompany(c); setView('form'); }} className="w-full bg-slate-50 p-5 rounded-xl flex items-center justify-between hover:bg-blue-600 hover:text-white transition-all">
                                        <span className="font-black uppercase text-xs">{c.name}</span>
                                        <span className="font-bold">➔</span>
                                    </button>
                                ))}
                            </div>
                        </div>
                        <button onClick={() => setView('admin-login')} className="mt-12 text-[10px] font-black text-slate-300 uppercase italic">Acesso Administrativo</button>
                    </div>
                </div>
            );

            if (view === 'form') return (
                <div className="min-h-screen bg-slate-100 flex items-center justify-center p-4">
                    <div className="max-w-xl w-full bg-white rounded-[2.5rem] shadow-2xl overflow-hidden text-left">
                        <header className="bg-slate-900 p-8 text-white flex justify-between items-center text-left leading-none">
                            <div className="text-left"><h2 className="text-xl font-black uppercase text-left">{activeCompany.name}</h2><p className="text-[9px] text-blue-400 font-bold uppercase mt-1">Protocolo Ético de SST</p></div>
                            <button onClick={() => setView('welcome')} className="text-white text-2xl">✕</button>
                        </header>
                        <form onSubmit={handleReport} className="p-8 space-y-6">
                            <div className="grid grid-cols-2 gap-4">
                                <div><label className="text-[10px] font-black text-slate-400 mb-1 block">Unidade *</label><input name="unidade" required className="input-rch" /></div>
                                <div><label className="text-[10px] font-black text-slate-400 mb-1 block">Data *</label><input type="date" name="dataOcorrencia" required className="input-rch" /></div>
                            </div>
                            <div className="bg-slate-50 p-6 rounded-2xl space-y-4">
                                <p className="text-[10px] font-black text-slate-400 uppercase">Identidade</p>
                                <div className="flex gap-2">
                                    <button type="button" onClick={() => setIsIdentified(false)} className={`flex-1 py-3 rounded-xl font-black text-[10px] ${!isIdentified ? 'bg-blue-600 text-white' : 'bg-white text-slate-400'}`}>Anónimo</button>
                                    <button type="button" onClick={() => setIsIdentified(true)} className={`flex-1 py-3 rounded-xl font-black text-[10px] ${isIdentified ? 'bg-blue-600 text-white' : 'bg-white text-slate-400'}`}>Identificado</button>
                                </div>
                                {isIdentified && <input name="nomeRelator" required className="input-rch bg-white" placeholder="Seu Nome Completo" />}
                            </div>
                            <textarea name="relato" required className="input-rch h-48 normal-case font-medium text-sm" placeholder="O Seu Relato Ético..."></textarea>
                            <button type="submit" disabled={loading} className="btn-rch py-5 text-lg shadow-xl">{loading ? "Processando..." : "Protocolar Denúncia"}</button>
                        </form>
                    </div>
                </div>
            );

            if (view === 'success') return (
                <div className="h-screen flex flex-col items-center justify-center text-center p-10 bg-slate-900 text-white">
                    <CheckCircle size={80} className="text-emerald-500 mb-6" />
                    <h2 className="text-4xl font-black uppercase mb-4">Recebido</h2>
                    <div className="bg-white p-10 rounded-[2.5rem] mb-12 shadow-2xl">
                        <p className="text-[10px] text-slate-400 mb-2 uppercase font-black">Código RCH</p>
                        <p className="text-5xl font-black text-blue-600 tracking-tighter">{protocol}</p>
                    </div>
                    <button onClick={() => window.location.reload()} className="btn-rch max-w-xs py-5">Sair</button>
                </div>
            );

            if (view === 'admin-login') return (
                <div className="h-screen flex items-center justify-center p-6 bg-slate-900">
                    <div className="max-w-xs w-full glass-card text-center">
                        <h2 className="text-xl font-black uppercase mb-10">Painel Auditora</h2>
                        <input type="password" className="input-rch text-center text-4xl mb-10 font-black" placeholder="••••" value={adminPass} onChange={e => setAdminPass(e.target.value)} />
                        <button onClick={() => adminPass === 'df2026' ? setView('admin-dash') : alert("Errada")} className="btn-rch py-4">Aceder</button>
                        <button onClick={() => setView('welcome')} className="mt-8 text-[9px] font-black text-slate-400 uppercase italic">Voltar</button>
                    </div>
                </div>
            );

            if (view === 'admin-dash') return (
                <div className="min-h-screen bg-slate-50 text-slate-900 p-6 text-left">
                    <header className="max-w-6xl mx-auto flex justify-between items-center mb-10 no-print text-left">
                        <div className="text-left leading-none">
                            <h2 className="text-xl font-black uppercase text-left leading-none">Gestão RCH 360</h2>
                            <p className="text-[10px] text-blue-600 font-bold uppercase mt-2 tracking-widest text-left">Independente Daiana Freitas</p>
                        </div>
                        <div className="flex flex-wrap justify-end gap-2">
                             <button onClick={() => setAdminTab('reports')} className={`px-4 py-2 rounded-xl font-black text-[9px] uppercase ${adminTab === 'reports' ? 'bg-blue-600 text-white' : 'bg-white'}`}>Relatos</button>
                             <button onClick={() => setAdminTab('folders')} className={`px-4 py-2 rounded-xl font-black text-[9px] uppercase ${adminTab === 'folders' ? 'bg-blue-600 text-white' : 'bg-white'}`}>Pastas</button>
                             <button onClick={() => setAdminTab('companies')} className={`px-4 py-2 rounded-xl font-black text-[9px] uppercase ${adminTab === 'companies' ? 'bg-blue-600 text-white' : 'bg-white'}`}>Empresas</button>
                             <button onClick={() => setView('welcome')} className="bg-red-50 text-red-600 px-4 py-2 rounded-xl font-black text-[9px] uppercase">Sair</button>
                        </div>
                    </header>

                    <main className="max-w-6xl mx-auto">
                        {adminTab === 'folders' && (
                            <div className="bg-white p-8 rounded-3xl shadow-sm text-left">
                                <h3 className="text-[10px] font-black uppercase text-slate-400 mb-6">Criar Pasta de Trabalho</h3>
                                <form onSubmit={async (e) => {
                                    e.preventDefault();
                                    const name = e.target.fn.value.toUpperCase();
                                    await db.collection(`${DATA_PATH}/folders`).add({ name });
                                    e.target.reset();
                                }} className="flex gap-4 mb-10">
                                    <input name="fn" required className="flex-1 input-rch" placeholder="NOME DA PASTA" />
                                    <button type="submit" className="bg-emerald-600 text-white px-10 py-4 rounded-xl font-black text-[11px] uppercase">Criar</button>
                                </form>
                                <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
                                    {folders.map(f => (
                                        <div key={f.id} className="bg-slate-50 p-5 rounded-2xl flex justify-between items-center group">
                                            <div className="flex items-center gap-3"><IconFolder size={18} /><span className="text-[10px] font-black uppercase">{f.name}</span></div>
                                            <button onClick={() => db.collection(`${DATA_PATH}/folders`).doc(f.id).delete()} className="text-red-500 opacity-0 group-hover:opacity-100">✕</button>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        )}

                        {adminTab === 'companies' && (
                            <div className="bg-white p-8 rounded-3xl shadow-sm text-left">
                                <h3 className="text-[10px] font-black uppercase text-slate-400 mb-8">Cadastrar Novo Cliente</h3>
                                <form onSubmit={async (e) => {
                                    e.preventDefault();
                                    const fd = new FormData(e.target);
                                    await db.collection(`${DATA_PATH}/companies`).add({
                                        name: fd.get('nome').toUpperCase(),
                                        dataTreino: fd.get('dataTreino'),
                                        vencimento: fd.get('vencimento')
                                    });
                                    e.target.reset();
                                }} className="grid grid-cols-1 md:grid-cols-3 gap-6">
                                    <div><label className="text-[9px] font-black text-slate-400 block mb-2">Empresa</label><input name="nome" required className="input-rch" /></div>
                                    <div><label className="text-[9px] font-black text-slate-400 block mb-2">Treino</label><input type="date" name="dataTreino" required className="input-rch" /></div>
                                    <div><label className="text-[9px] font-black text-slate-400 block mb-2">Validade</label><input type="date" name="vencimento" required className="input-rch" /></div>
                                    <button type="submit" className="md:col-span-3 btn-rch py-4">Ativar Canal</button>
                                </form>
                                <div className="grid grid-cols-1 md:grid-cols-3 gap-6 mt-12">
                                    {companies.map(c => (
                                        <div key={c.id} className="bg-slate-50 p-6 rounded-3xl border relative group">
                                            <p className="font-black uppercase text-sm mb-4">{c.name}</p>
                                            <div className="flex justify-between items-center text-[10px] font-bold">
                                                <button onClick={() => genQR(c)} className="text-blue-600 uppercase">QR Code</button>
                                                <button onClick={() => db.collection(`${DATA_PATH}/companies`).doc(c.id).delete()} className="text-red-500 uppercase">Eliminar</button>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            </div>
                        )}

                        {adminTab === 'reports' && (
                            <div className="space-y-8">
                                <button onClick={() => window.print()} className="bg-slate-900 text-white px-8 py-4 rounded-xl font-black text-[10px] uppercase no-print shadow-xl">Imprimir Relatórios PDF</button>
                                {companies.map(emp => {
                                    const empReports = reports.filter(r => r.empresaId === emp.id || r.empresaName === emp.name);
                                    if(empReports.length === 0) return null;
                                    return (
                                        <div key={emp.id} className="bg-white rounded-3xl border p-10 text-left report-box">
                                            <h3 className="text-3xl font-black uppercase italic mb-8 border-b pb-6 leading-none text-left">{emp.name}</h3>
                                            <div className="space-y-10">
                                                {empReports.map(r => (
                                                    <div key={r.id} className="p-10 bg-slate-50 rounded-[3rem] border-l-[25px] border-blue-600 relative text-left">
                                                        <div className="flex justify-between items-start mb-8 text-left leading-none">
                                                            <div className="text-left"><p className="text-[10px] font-black text-blue-600 uppercase leading-none mb-2">Protocolo: {r.protocolo}</p><p className="text-[9px] opacity-40 uppercase leading-none">Abertura: {new Date(r.dataProtocolo).toLocaleDateString()}</p></div>
                                                            <div className="no-print"><button onClick={() => db.collection(`${DATA_PATH}/reports`).doc(r.id).delete()} className="text-red-500 font-bold uppercase text-[10px]">Apagar</button></div>
                                                        </div>
                                                        <div className="grid md:grid-cols-2 gap-8 mb-8 text-left">
                                                            <div className="bg-white p-6 rounded-2xl">
                                                                <p className="text-[9px] font-black uppercase text-slate-400 mb-4">Abertura Técnica</p>
                                                                <input className="input-rch h-12 py-0 text-xs shadow-inner no-print" placeholder="Nº Ocorrência" defaultValue={r.numOcorrencia} onBlur={(e) => updateRep(r.id, { numOcorrencia: e.target.value })} />
                                                                <p className="print-only text-sm font-bold">Ocorrência: {r.numOcorrencia || 'Pendente'}</p>
                                                            </div>
                                                            <div className="bg-white p-6 rounded-2xl"><p className="text-[9px] font-black uppercase text-slate-400 mb-4">Depoimento Auditado</p><p className="text-xs italic font-medium leading-relaxed">"{r.relato}"</p></div>
                                                        </div>
                                                        <div className="bg-white p-6 rounded-2xl no-print text-left">
                                                            <p className="text-[9px] font-black uppercase text-emerald-600 mb-4">Fechamento e Parecer</p>
                                                            <textarea className="input-rch h-24 py-4 text-xs shadow-inner normal-case" placeholder="Conclusão da Auditoria..." defaultValue={r.conclusaoTecnica} onBlur={(e) => updateRep(r.id, { conclusaoTecnica: e.target.value })}></textarea>
                                                        </div>
                                                        <div className="print-only mt-10 pt-8 border-t border-slate-900 flex justify-between">
                                                            <div className="signature-line">Daiana Freitas (RCH)</div>
                                                            <div className="signature-line">Empresa Cliente</div>
                                                        </div>
                                                    </div>
                                                ))}
                                            </div>
                                        </div>
                                    );
                                })}
                            </div>
                        )}
                    </main>

                    {qrComp && (
                        <div className="fixed inset-0 bg-slate-900/95 flex items-center justify-center p-6 z-50 no-print">
                            <div className="bg-white p-12 rounded-[3rem] text-center animate-fade">
                                <h3 className="text-xl font-black uppercase italic mb-10">{qrComp.name}</h3>
                                <div id="qr-area" className="mb-10 inline-block border-8 border-slate-50 p-4 rounded-3xl"></div>
                                <div className="flex gap-4"><button onClick={() => window.print()} className="btn-rch bg-slate-900">Imprimir</button><button onClick={() => setQrComp(null)} className="btn-rch bg-red-50 text-red-600">Fechar</button></div>
                            </div>
                        </div>
                    )}
                </div>
            );

            return null;
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
